#!/bin/bash

# 1. 탐색할 대상 파일들의 경로 패턴 설정
# (2026년도 데이터를 대상으로 함)
TARGET_FILES="/data/DataStory/report/*/2026_Y/basic/0/summary.txt"

# 2. 출력 결과물의 헤더(머리글) 디자인
echo "============================================="
echo -e "사이트(디렉토리)\t|\t페이지뷰(PV)"
echo "============================================="

# 3. awk를 이용한 데이터 추출 및 합계 계산
awk '
BEGIN {
    # 합계 변수 초기화
    total_sum = 0
}

# 4. "#"으로 시작하지 않는 데이터 라인을 찾음
!/^#/ {
    # 첫 번째 컬럼(페이지뷰)의 값을 total_sum에 누적
    total_sum += $1
    
    # FILENAME(현재 처리 중인 파일의 절대경로)에서 디렉토리 번호 추출
    # 예: /data/DataStory/report/00009/2026_Y/... -> "/"를 기준으로 나누면 5번째가 00009
    split(FILENAME, path, "/")
    site_id = path[5]
    
    # 5. 개별 디렉토리명과 PV 값을 줄 맞춰 출력
    printf " %-12s \t|\t %d\n", site_id, $1
    
    # 속도 최적화: 값을 찾았으므로 현재 파일의 나머지 줄은 무시하고 다음 파일로 즉시 이동
    nextfile
}

# 6. 모든 파일 탐색이 종료된 후 마지막에 실행되는 영역
END {
    print "---------------------------------------------"
    # 누적된 총 합계 출력
    printf " [총 합계]    \t|\t %d\n", total_sum
    print "============================================="
}' $TARGET_FILES
