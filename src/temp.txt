// 1. div[role="rowgroup"] 태그 아래의 모든 행(row)을 선택합니다.
const rows = document.querySelectorAll('div[role="rowgroup"] div[role="row"]');
const resultList = [];

rows.forEach((row) => {
  // 행마다 데이터를 저장할 객체
  let rowData = {
    name: null,       // aria-colindex="2" (이름)
    empId: null,      // aria-colindex="4" (사번)
    role: null,       // aria-colindex="5" (역할)
    lastLogin: null   // aria-colindex="6" (마지막 로그인)
  };

  // --- 2번: 이름 (aria-colindex="2") ---
  // 하위의 a 태그를 찾아 innerText 또는 텍스트 콘텐츠를 가져옵니다.
  // (구조상 a 태그 내부에 이름이 텍스트로 들어가 있습니다)
  const nameCell = row.querySelector('div[aria-colindex="2"]');
  if (nameCell) {
    const aTag = nameCell.querySelector('a');
    if (aTag) {
      rowData.name = aTag.innerText.trim();
    }
  }

  // --- 4번: 사번 (aria-colindex="4") ---
  // 최하위 span을 찾아 title 속성 값을 가져옵니다.
  const empIdCell = row.querySelector('div[aria-colindex="4"]');
  if (empIdCell) {
    const spans = empIdCell.querySelectorAll('span');
    if (spans.length > 0) {
      const lastSpan = spans[spans.length - 1];
      rowData.empId = lastSpan.getAttribute('title') || lastSpan.innerText.trim();
    }
  }

  // --- 5번: Role (aria-colindex="5") ---
  // 최하위 span을 찾아 title 속성 값을 가져옵니다.
  const roleCell = row.querySelector('div[aria-colindex="5"]');
  if (roleCell) {
    const spans = roleCell.querySelectorAll('span');
    if (spans.length > 0) {
      const lastSpan = spans[spans.length - 1];
      rowData.role = lastSpan.getAttribute('title') || lastSpan.innerText.trim();
    }
  }

  // --- 6번: 마지막 로그인 (aria-colindex="6") ---
  // 최하위 span을 찾아 title이 아닌 span 태그 내부의 텍스트를 가져옵니다.
  const loginCell = row.querySelector('div[aria-colindex="6"]');
  if (loginCell) {
    const spans = loginCell.querySelectorAll('span');
    if (spans.length > 0) {
      const lastSpan = spans[spans.length - 1];
      rowData.lastLogin = lastSpan.innerText.trim();
    }
  }

  // 필요한 데이터가 하나라도 존재하는 행만 결과 리스트에 추가합니다.
  if (rowData.name || rowData.empId || rowData.role || rowData.lastLogin) {
    resultList.push(rowData);
  }
});

// 결과를 테이블 형태로 콘솔에 깔끔하게 출력합니다.
console.table(resultList);
